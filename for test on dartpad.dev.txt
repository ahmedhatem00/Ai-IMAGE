// ========================================================
// 🌐 Pollinations API Service
// ========================================================
class PollinationsService {
  final String _apiKey;
  final String _referrer;

  PollinationsService({String? apiKey, String? referrer})
      : _apiKey = apiKey ?? '',
        _referrer = referrer ?? 'flutter-app';

  Future<Uint8List> generateImage({
    required String prompt,
    String model = 'flux',
    String style = 'none',
    int width = 1024,
    int height = 1024,
    String negativePrompt = '',
    int? seed,
  }) async {
    String fullPrompt = prompt.trim();
    if (style.isNotEmpty && style != 'none') {
      final styleName = availableStyles[style] ?? style;
      fullPrompt = '$fullPrompt, $styleName style';
    }
    fullPrompt = '$fullPrompt, highly detailed, sharp focus, high quality';
    if (negativePrompt.isNotEmpty) {
      fullPrompt = '$fullPrompt, avoid: $negativePrompt';
    }

    final encodedPrompt = Uri.encodeComponent(fullPrompt);
    var url = Uri.parse('$baseUrl$encodedPrompt');
    final params = <String, String>{
      'width': width.toString(),
      'height': height.toString(),
      'model': model,
      'enhance': 'true',
      'nologo': 'true',
      'referrer': _referrer,
      if (seed != null) 'seed': seed.toString(),
    };
    url = url.replace(queryParameters: params);

    final headers = <String, String>{
      'Accept': 'image/*',
      if (_apiKey.isNotEmpty) 'Authorization': 'Bearer $_apiKey',
    };

    final response = await http.get(url, headers: headers).timeout(
          const Duration(seconds: 120),
          onTimeout: () => throw TimeoutException('انتهت مهلة الطلب / Request timed out'),
        );

    final contentType = response.headers['content-type'] ?? '';
    if (response.statusCode == 200 &&
        response.bodyBytes.length > 1024 &&
        contentType.contains('image')) {
      return response.bodyBytes;
    }
    throw Exception('فشل التوليد (HTTP ${response.statusCode})');
  }

  static Future<Size> getImageSize(Uint8List bytes) async {
    final codec = await ui.instantiateImageCodec(bytes);
    final frame = await codec.getNextFrame();
    final size = Size(
      frame.image.width.toDouble(),
      frame.image.height.toDouble(),
    );
    frame.image.dispose();
    return size;
  }
}

/// ينفذ مجموعة من المهام مع حد أقصى لعدد الطلبات المتزامنة
/// (بدلاً من إرسالها كلها دفعة واحدة أو تسلسلياً واحدة تلو الأخرى).
Future<List<T>> runWithConcurrencyLimit<T>(
  List<Future<T> Function()> tasks, {
  int limit = 3,
}) async {
  final results = List<T?>.filled(tasks.length, null);
  int nextIndex = 0;

  Future<void> worker() async {
    while (true) {
      final i = nextIndex;
      if (i >= tasks.length) return;
      nextIndex++;
      results[i] = await tasks[i]();
    }
  }

  final workers = List.generate(
    math.min(limit, tasks.length),
    (_) => worker(),
  );
  await Future.wait(workers);
  return results.cast<T>();
}

// 🧩 Mosaic Generator
// ========================================================
int _gridColumns(int count) => math.sqrt(count).ceil();

Future<Uint8List> combineImages(List<Uint8List> images, int columns) async {
  if (images.isEmpty) throw Exception('لا توجد صور لدمجها');
  final rows = (images.length / columns).ceil();

  final decoded = <ui.Image>[];
  try {
    for (final bytes in images) {
      final codec = await ui.instantiateImageCodec(bytes);
      final frame = await codec.getNextFrame();
      decoded.add(frame.image);
    }

    const tileSize = 1024;
    final canvasWidth = columns * tileSize;
    final canvasHeight = rows * tileSize;

    final recorder = ui.PictureRecorder();
    final canvas = Canvas(recorder);
    final paint = Paint()..filterQuality = FilterQuality.high;

    for (int i = 0; i < decoded.length; i++) {
      final img = decoded[i];
      final x = (i % columns) * tileSize;
      final y = (i ~/ columns) * tileSize;
      canvas.drawImageRect(
        img,
        Rect.fromLTWH(0, 0, img.width.toDouble(), img.height.toDouble()),
        Rect.fromLTWH(
          x.toDouble(),
          y.toDouble(),
          tileSize.toDouble(),
          tileSize.toDouble(),
        ),
        paint,
      );
    }

    final picture = recorder.endRecording();
    final combined = await picture.toImage(canvasWidth, canvasHeight);
    try {
      final byteData =
          await combined.toByteData(format: ui.ImageByteFormat.png);
      if (byteData == null) throw Exception('فشل ترميز صورة الموزاييك');
      return byteData.buffer.asUint8List();
    } finally {
      combined.dispose();
    }
  } finally {
    // تجنب تسريب الذاكرة: تحرير كل الصور المفكوكة بعد الاستخدام
    for (final img in decoded) {
      img.dispose();
    }
  }
}

/// يقتصّ رسالة الخطأ بأمان دون التسبب بخطأ RangeError
/// إذا كانت الرسالة أقصر من الطول المطلوب.
String truncate(String text, int maxLen) {
  if (text.length <= maxLen) return text;
  return '${text.substring(0, maxLen)}...';
}

class _GeneratePageState extends State<GeneratePage> {
  final _promptController = TextEditingController();
  final _negativeController = TextEditingController();

  @override
  void initState() {
    super.initState();
    final state = context.read<AppState>();
    _promptController.text = state.prompt;
    _negativeController.text = state.negativePrompt;
    _promptController.addListener(() {
      context.read<AppState>().setPrompt(_promptController.text);
    });
    _negativeController.addListener(() {
      context.read<AppState>().setNegativePrompt(_negativeController.text);
    });
  }

  @override
  void dispose() {
    _promptController.dispose();
    _negativeController.dispose();
    super.dispose();
  }

  Future<void> _generateImages(BuildContext ctx, {bool isMosaic = false}) async {
    final state = ctx.read<AppState>();
    final prompt = _promptController.text.trim();
    if (prompt.isEmpty) {
      ScaffoldMessenger.of(ctx).showSnackBar(
        SnackBar(content: Text('${tr(ctx, 'prompt')} ${tr(ctx, 'error')}')),
      );
      return;
    }

    state.setLoading(true);
    final service = PollinationsService(apiKey: apiKey, referrer: referrer);
    final dims = shapeDimensions[state.shape]!;
    final width = dims[0];
    final height = dims[1];
    final count = isMosaic ? state.mosaicTiles : state.imageCount;
    final negative = _negativeController.text.trim();
    final baseSeed = DateTime.now().millisecondsSinceEpoch % 1000000;

    try {
      // 🚀 الطلبات تُرسل بشكل متزامن (بحد أقصى 3 في نفس الوقت) بدلاً من
      // تسلسلي واحدة تلو الأخرى، مما يسرّع التوليد بشكل ملحوظ خصوصاً
      // في وضع الموزاييك (عدة صور دفعة واحدة).
      final tasks = List.generate(
        count,
        (i) => () => service.generateImage(
              prompt: prompt,
              model: state.model,
              style: state.style,
              width: width,
              height: height,
              negativePrompt: negative,
              seed: baseSeed + i,
            ),
      );
      final imageBytes = await runWithConcurrencyLimit(tasks, limit: 3);

      Uint8List finalBytes;
      int finalWidth, finalHeight;
      if (isMosaic) {
        final cols = _gridColumns(count);
        finalBytes = await combineImages(imageBytes, cols);
        final rows = (count / cols).ceil();
        finalWidth = cols * 1024;
        finalHeight = rows * 1024;
      } else {
        // الأبعاد معروفة مسبقاً من الطلب نفسه، لا داعي لإعادة فك تشفير
        // الصورة فقط لقراءة العرض/الارتفاع (كان هذا يبطئ الأداء بلا فائدة).
        finalBytes = imageBytes.first;
        finalWidth = width;
        finalHeight = height;
      }

      if (!ctx.mounted) return;

      final generated = GeneratedImage(
        bytes: finalBytes,
        prompt: prompt,
        model: state.model,
        style: state.style,
        shape: state.shape,
        timestamp: DateTime.now(),
        width: finalWidth,
        height: finalHeight,
      );
      state.addHistory(generated);

      ScaffoldMessenger.of(ctx).showSnackBar(
        SnackBar(
          content: Text(isMosaic
              ? '✅ ${tr(ctx, 'mosaic')} ${tr(ctx, 'generate')}'
              : '✅ ${tr(ctx, 'generate')}'),
        ),
      );
    } catch (e) {
      state.setError(e.toString());
      if (!ctx.mounted) return;
      ScaffoldMessenger.of(ctx).showSnackBar(
        SnackBar(
          content: Text('${tr(ctx, 'error')}: ${truncate(e.toString(), 100)}'),
        ),
      );
    } finally {
      state.setLoading(false);
    }
  }

class HistoryPage extends StatelessWidget {
  const HistoryPage({super.key});

  @override
  Widget build(BuildContext ctx) {
    final state = ctx.watch<AppState>();
    return Scaffold(
      appBar: AppBar(
        title: Text(tr(ctx, 'history')),
        actions: [
          if (state.history.isNotEmpty)
            IconButton(
              icon: const Icon(Icons.delete_sweep),
              onPressed: () {
                showDialog(
                  context: ctx,
                  builder: (dialogCtx) => AlertDialog(
                    title: Text(tr(ctx, 'clear_history')),
                    content: Text(tr(ctx, 'are_you_sure')),
                    actions: [
                      TextButton(
                        onPressed: () => Navigator.pop(dialogCtx),
                        child: Text(tr(ctx, 'cancel')),
                      ),
                      FilledButton(
                        onPressed: () {
                          state.clearHistory();
                          Navigator.pop(dialogCtx);
                        },
                        style: FilledButton.styleFrom(backgroundColor: Colors.red),
                        child: Text(tr(ctx, 'clear_history')),
                      ),
                    ],
                  ),
                );
              },
            ),
        ],
      ),
      body: state.history.isEmpty
          ? Center(child: Text(tr(ctx, 'no_images')))
          : ListView.builder(
              padding: const EdgeInsets.all(8),
              itemCount: state.history.length,
              itemBuilder: (_, i) {
                final img = state.history[i];
                return Card(
                  margin: const EdgeInsets.symmetric(vertical: 6),
                  child: ListTile(
                    leading: SizedBox(
                      width: 60,
                      height: 60,
                      child: Image.memory(
                        img.bytes,
                        fit: BoxFit.cover,
                        gaplessPlayback: true,
                      ),
                    ),
                    title: Text(
                      truncate(img.prompt, 40),
                      maxLines: 1,
                      overflow: TextOverflow.ellipsis,
                    ),
                    subtitle: Text(
                      '${DateFormat.yMd().add_jm().format(img.timestamp)}  •  ${img.dimensions}',
                    ),
                    onTap: () {
                      showDialog(
                        context: ctx,
                        builder: (dialogCtx) => AlertDialog(
                          content: Column(
                            mainAxisSize: MainAxisSize.min,
                            children: [
                              Image.memory(img.bytes, fit: BoxFit.contain),
                              const SizedBox(height: 8),
                              Text(img.prompt),
                              Text(
                                '${img.dimensions}  •  ${DateFormat.yMd().add_jm().format(img.timestamp)}',
                              ),
                            ],
                          ),
                          actions: [
                            TextButton(
                              onPressed: () => Navigator.pop(dialogCtx),
                              child: Text(tr(ctx, 'close')),
                            ),
                          ],
                        ),
                      );
                    },
                  ),
                );
              },
            ),
    );
  }
}
